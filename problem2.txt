def tower_of_hanoi(n, source, target, middle):
    if n == 1:
        print(f"Move disk 1 from {source} to {target}")
        return
    
    tower_of_hanoi(n-1, source, middle, target)
   
    print(f"Move disk {n} from {source} to {target}")
   
    tower_of_hanoi(n-1, middle, target, source)


n = 3
tower_of_hanoi(n, 'A', 'C', 'B')
